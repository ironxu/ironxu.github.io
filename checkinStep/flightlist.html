飞友网络 值机接口

获取航段(FlightList)

调用方式

posts = [
    'airline' => airline,         // 需要请求的机场，例如'3U'
    'action'  => 'FlightList',    // FlightList 用于获取航段
    // 根据Entrance返回的数据，处理拼接的数据
    'body'    => 'passengerName=%E9%9F%A9%E7%A7%80%E5%A8%9F&idType=0+%E8%BA%AB%E4%BB%BD%E8%AF%81&idNo=230105198403150043&mobileNo=18071533938&depCity=XMN&depCityCode=XMN&orgId=3UAIR&airlineCode=3U&userID=3U&language=CN',
    'useid'   => 'c1aaf79a3b284681a9b41b4df8cbb1f1',
];
posts = json_encode(posts);       // 参数进行 json encode
result = http_post(url, posts);   // HTTP post方法调用接口
